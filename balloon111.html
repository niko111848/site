const colors = ['red', 'orange', 'yellow', 'green', 'black', 'blue', 'purple']; function createBalloon() { const balloon = { color: colors[Math.floor(Math.random() * colors.length)], x: Math.random() * canvas.width, y: canvas.height, diameter: 20, speed: Math.random() * 3 + 1, burst: false }; balloons.push(balloon); } function updateBalloons() { balloons.forEach((balloon, index) => { if (!balloon.burst) { balloon.y -= balloon.speed; balloon.diameter += 0.1; if (balloon.y < canvas.height / 3) { balloon.burst = true; setTimeout(() => { balloons.splice(index, 1); }, index * 500); } } }); }
